select n.newsId, n.date, n.text, c.commentDate, c.text from
              News n
          inner join
              Comments c on n.newsId = c.newsId
          inner join
(select max(Comments.commentDate) lastDate, Comments.newsId
from Comments
group by Comments.newsId) lastComment on c.newsId = lastComment.newsId and c.commentDate = lastComment.lastDate
order by n.date desc limit 10
